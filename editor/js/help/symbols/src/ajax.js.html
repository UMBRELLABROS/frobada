1 // Transfer data to server and read infos
  2 
  3 /**
  4  * Ajax handle PUT, DELETE, GET and POST
  5  *
  6  */
  7 var Ajax = function() {
  8 	// private
  9 	var client = null;
 10 	var callbackFunction; // callback
 11 	var async = true;
 12 	var responseType = '';
 13 	// public
 14 	this.URL = ''; // path to server
 15 
 16 	// pupblic functions
 17 	this.post = postData; //post request
 18 	this.get = getData; //get request
 19 	this.put = putData; //put request
 20 	this.delete = deleteData; //delete request
 21 
 22 	this.returnFunction = setCallback; // callback setzen
 23 
 24 	// call with 2 parameters
 25 	if (this instanceof arguments.callee) {
 26 		init.apply(this, arguments);
 27 	}
 28 	function init() {
 29 		this.URL = arguments[0];
 30 		this.returnFunction(arguments[1]);
 31 	}
 32 
 33 	if (window.XMLHttpRequest) {
 34 		client = new XMLHttpRequest();
 35 	} else if (window.ActiveXObject) {
 36 		client = new ActiveXObject('Microsoft.XMLHTTP');
 37 	}
 38 	client.responseType = responseType;
 39 
 40 	// request and error handling
 41 	client.onreadystatechange = function() {
 42 		if (client.readyState == 4 && client.status == 200) {
 43 			res = client.responseText; // result
 44 			callbackFunction(res); // send data to callback
 45 		}
 46 	};
 47 
 48 	// definition
 49 	// https://www.mediaevent.de/javascript/XmlHttp-Request-Methoden-Eigenschaften.html
 50 
 51 	/**
 52 	 * post request
 53 	 * @param {Object} formData - FormData from <form
 54 	 *
 55 	 */
 56 	function postData(formData) {
 57 		send(this.URL, 'post', formData);
 58 	}
 59 
 60 	/**
 61 	 * send a put request
 62 	 * @param {string} formData
 63 	 */
 64 	function putData(formData) {
 65 		send(this.URL, 'put', formData);
 66 	}
 67 
 68 	// get request
 69 	function getData() {
 70 		send(this.URL, 'get');
 71 	}
 72 
 73 	// delete request
 74 	function deleteData() {
 75 		send(this.URL, 'delete');
 76 	}
 77 
 78 	function send(url, type, payload) {
 79 		client.open(type, url, async);
 80 		if (payload) {
 81 			client.send(payload);
 82 		} else {
 83 			client.send();
 84 		} /* Send to server */
 85 	}
 86 
 87 	// store callback
 88 	function setCallback(callback) {
 89 		callbackFunction = callback;
 90 	}
 91 };
