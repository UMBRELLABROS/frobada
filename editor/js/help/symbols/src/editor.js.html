 70 function scanDOM(source, elements) {
 71   var nodes = source.childNodes;
 72   nodes.forEach(element => {
 73     if (element.nodeName == "DIV") {
 74       elements.push(element);
 75       if (element.childNodes.length > 0) {
 76         elements = scanDOM(element, elements);
 77       }
 78     }
 79   });
 80   return elements;
 81 }
 82 
 83 function scanDOMStructure(source, structure) {
 84   var nodes = source.childNodes;
 85   nodes.forEach(element => {
 86     if (element.nodeName == "DIV") {
 87       var objDIV = {
 88         "nodeName": "DIV",
 89         "id": element.id,
 90         "childNodes": []
 91       }
 92       structure.push(objDIV);
 93       if (element.childNodes.length > 0) {
 94         objDIV.childNodes = (scanDOMStructure(element, []));
 95       }
 96     }
 97   });
 98   return structure;
 99 }

101 function colorToHex(color) {
102   // Convert any CSS color to a hex representation
103   // Examples:
104   // colorToHex('red')            # '#ff0000'
105   // colorToHex('rgb(255, 0, 0)') # '#ff0000'
106   var rgba = colorToRGBA(color);
107   var hex = [0, 1, 2].map(
108     function (idx) { return byteToHex(rgba[idx]); }
109   ).join('');
110   return "#" + hex;
