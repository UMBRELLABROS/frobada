1 /** Layout
  2  *
  3  * Write the elements and add the styles
  4  * loadStructure
  5  * loadStyles
  6  */
  7 var Layout = function() {
  8 	this.loadStructure = loadStructure;
  9 	this.loadStyles = loadStyles;
 10 
 11 	function loadStructure(id, json) {
 12 		var data = JSON.parse(json);
 13 		writeElements($(id), data);
 14 	}
 15 
 16 	function loadStyles(json) {
 17 		var data = JSON.parse(json);
 18 		writeStyles(data);
 19 	}
 20 
 21 	/** load the array of HTML elements */
 22 	function writeElements(obj, elements) {
 23 		elements.map(element => {
 24 			writeElement(obj, element);
 25 			// handle childNodes
 26 			writeElements($(element.id), element.childNodes);
 27 		});
 28 	}
 29 
 30 	/** write one HTML element and attach to parent */
 31 	function writeElement(obj, element) {
 32 		var item = dCE(element.nodeName);
 33 		item.id = element.id;
 34 		obj.appendChild(item);
 35 	}
 36 
 37 	/** load the array of styles */
 38 	function writeStyles(elements) {
 39 		elements.map(element => {
 40 			writeStyle($(element.id), element.attributes);
 41 		});
 42 	}
 43 
 44 	/** write the style */
 45 	function writeStyle(element, styles) {
 46 		for ([key, value] of Object.entries(styles)) {
 47 			if (typeof value === 'object') {
 48 				// key == style
 49 				writeStyle(element[key], value);
 50 			} else {
 51 				// key == style attribute
 52 				element[key] = value;
 53 			}
 54 		}
 55 	}
 56 };
