1 /** Layout
  2  *
  3  * Write the elements and andd the styles
  4  */
  5 var Layout = function() {
  6 	this.loadStructure = loadStructure;
  7 
  8 	function loadStructure(id, json) {
  9 		var data = JSON.parse(json);
 10 		writeElements($(id), data);
 11 	}
 12 
 13 	/** load the array of HTML elements */
 14 	function writeElements(obj, elements) {
 15 		elements.map(element => {
 16 			writeElement(obj, element);
 17 			// handle childNodes
 18 			writeElements($(element.id), element.childNodes);
 19 		});
 20 	}
 21 
 22 	/** write one HTML element and attach to parent */
 23 	function writeElement(obj, element) {
 24 		var item = dCE(element.nodeName);
 25 		item.id = element.id;
 26 		obj.appendChild(item);
 27 	}
 28 };
