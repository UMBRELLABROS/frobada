 69 	function scanDOM(source, elements) {
 70 		var nodes = source.childNodes;
 71 		nodes.forEach(element => {
 72 			if (element.nodeName == 'DIV') {
 73 				elements.push(element);
 74 				if (element.childNodes.length > 0) {
 75 					elements = scanDOM(element, elements);
 76 				}
 77 			}
 78 		});
 79 		return elements;
 80 	}
 81 
 82 	function scanDOMStructure(source, structure) {
 83 		var nodes = source.childNodes;
 84 		nodes.forEach(element => {
 85 			if (element.nodeName == 'DIV') {
 86 				var objDIV = {
 87 					nodeName: 'DIV',
 88 					id: element.id,
 89 					childNodes: []
 90 				};
 91 				structure.push(objDIV);
 92 				if (element.childNodes.length > 0) {
 93 					objDIV.childNodes = scanDOMStructure(element, []);
 94 				}
 95 			}
 96 		});
 97 		return structure;
 98 	}
