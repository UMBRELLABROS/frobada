1 /** data-attributes
  2  *
  3  * get and set the data-attributes
  4  */
  5 var DataAttribute = function() {
  6 	this.loadAttributes = loadAttributes;
  7 
  8 	function loadAttributes(json) {
  9 		var data = JSON.parse(json);
 10 		writeAttributes(data);
 11 	}
 12 
 13 	/** load the data-attributes if available */
 14 	function writeAttributes(elements) {
 15 		elements.map(element => {
 16 			if (element.data) {
 17 				writeAttribute($(element.id), element.data);
 18 			}
 19 		});
 20 	}
 21 
 22 	/** add data-attributes to the element */
 23 	function writeAttribute(element, data) {
 24 		for ([key, value] of Object.entries(data)) {
 25 			if (typeof value === 'object') {
 26 				setDataAttribute(element[key], value);
 27 			} else {
 28 				element.setAttribute('data-' + key, value);
 29 			}
 30 		}
 31 	}
 32 };
